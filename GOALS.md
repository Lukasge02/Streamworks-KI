# app/services/user_service.py
from enum import Enum
from typing import Dict, Any, List
from langchain.chains import ConversationalRetrievalChain
from langchain.agents import AgentType, initialize_agent
from langchain.memory import ConversationSummaryBufferMemory

class UserRole(Enum):
    DEVELOPER = "developer"
    ADMIN = "admin"
    BUSINESS_USER = "business_user"
    SUPPORT = "support"
    ARCHITECT = "architect"

class UserGroupService:
    def __init__(self, rag_service, mistral_service):
        self.rag_service = rag_service
        self.mistral_service = mistral_service
        self.chains = {}
        self.agents = {}
        self.initialize_chains()
    
    def initialize_chains(self):
        """Initialize different chains for different user groups"""
        
        # Developer Chain - Technical Documentation Focus
        self.chains[UserRole.DEVELOPER] = self.create_developer_chain()
        
        # Admin Chain - System Management Focus
        self.chains[UserRole.ADMIN] = self.create_admin_chain()
        
        # Business User Chain - High-Level Explanation Focus
        self.chains[UserRole.BUSINESS_USER] = self.create_business_chain()
        
        # Support Chain - Troubleshooting Focus
        self.chains[UserRole.SUPPORT] = self.create_support_chain()
        
        # Architect Chain - Design Pattern Focus
        self.chains[UserRole.ARCHITECT] = self.create_architect_chain()
    
    def create_developer_chain(self) -> ConversationalRetrievalChain:
        """Chain für Entwickler - Code-fokussiert"""
        return ConversationalRetrievalChain.from_llm(
            llm=self.mistral_service.get_llm(),
            retriever=self.rag_service.get_retriever(
                search_kwargs={"k": 5, "filter": {"category": ["code", "api", "technical"]}}
            ),
            memory=ConversationSummaryBufferMemory(
                llm=self.mistral_service.get_llm(),
                max_token_limit=1000,
                return_messages=True
            ),
            return_source_documents=True,
            verbose=True
        )
    
    def create_business_chain(self) -> ConversationalRetrievalChain:
        """Chain für Business User - High-Level Fokus"""
        return ConversationalRetrievalChain.from_llm(
            llm=self.mistral_service.get_llm(),
            retriever=self.rag_service.get_retriever(
                search_kwargs={"k": 3, "filter": {"category": ["business", "overview", "process"]}}
            ),
            memory=ConversationSummaryBufferMemory(
                llm=self.mistral_service.get_llm(),
                max_token_limit=500,
                return_messages=True
            ),
            return_source_documents=True
        )
    
    async def query_for_user_group(self, 
                                  user_role: UserRole, 
                                  query: str, 
                                  chat_history: List = None) -> Dict[str, Any]:
        """Query mit user-spezifischer Chain"""
        
        # Get appropriate chain
        chain = self.chains.get(user_role)
        if not chain:
            chain = self.chains[UserRole.BUSINESS_USER]  # Default fallback
        
        # Execute query
        result = await chain.arun(
            question=query,
            chat_history=chat_history or []
        )
        
        return {
            "answer": result["answer"],
            "sources": self.format_sources(result["source_documents"]),
            "user_role": user_role.value,
            "confidence": self.calculate_confidence(result)
        }
```

# app/processors/source_processors.py
from abc import ABC, abstractmethod
from typing import Dict, Any, List
import os
from pathlib import Path

class SourceProcessor(ABC):
    """Base class for source-specific processing"""
    
    @abstractmethod
    async def process_document(self, file_path: str, metadata: Dict[str, Any]) -> Dict[str, Any]:
        pass
    
    @abstractmethod
    def extract_metadata(self, file_path: str) -> Dict[str, Any]:
        pass

class StreamWorksProcessor(SourceProcessor):
    """Processor for StreamWorks help files"""
    
    async def process_document(self, file_path: str, metadata: Dict[str, Any]) -> Dict[str, Any]:
        """Process StreamWorks help document"""
        
        # Extract specific metadata
        streamworks_metadata = self.extract_metadata(file_path)
        
        # Enhanced metadata
        enhanced_metadata = {
            **metadata,
            "source_type": "streamworks_help",
            "source_path": file_path,
            "title": streamworks_metadata.get("title", Path(file_path).stem),
            "category": self.determine_category(file_path),
            "tags": self.extract_tags(file_path),
            "extra_metadata": {
                "module": streamworks_metadata.get("module"),
                "version": streamworks_metadata.get("version"),
                "last_updated": streamworks_metadata.get("last_updated")
            }
        }
        
        return enhanced_metadata
    
    def determine_category(self, file_path: str) -> str:
        """Determine category based on file path and content"""
        path_lower = file_path.lower()
        
        if "batch" in path_lower or "job" in path_lower:
            return "batch_processing"
        elif "xml" in path_lower or "template" in path_lower:
            return "xml_templates"
        elif "api" in path_lower:
            return "api_documentation"
        elif "admin" in path_lower:
            return "administration"
        else:
            return "general_help"

class JIRAProcessor(SourceProcessor):
    """Processor for JIRA documentation"""
    
    async def process_document(self, file_path: str, metadata: Dict[str, Any]) -> Dict[str, Any]:
        """Process JIRA document"""
        
        jira_metadata = self.extract_metadata(file_path)
        
        return {
            **metadata,
            "source_type": "jira",
            "source_path": file_path,
            "title": jira_metadata.get("title", Path(file_path).stem),

                "project_key": jira_metadata.get("project_key"),
                "issue_type": jira_metadata.get("issue_type"),

            }
        }

                "schema": ddds_metadata.get("schema"),

                "data_type": ddds_metadata.get("data_type")

# app/services/rag_service_v2.py
from typing import List, Dict, Any, Optional
from langchain.schema import Document

class RAGServiceV2(RAGService):
    """Enhanced RAG Service with source citations"""
    
    async def search_with_citations(self, 
                                   query: str, 
                                   k: int = 5,
                                   user_role: Optional[UserRole] = None) -> Dict[str, Any]:
        """Search with detailed source citations"""
        
        # Get relevant documents
        docs = await self.search_documents(query, k=k)
        
        # Generate response with citations
        response = await self.generate_response_with_citations(query, docs)
        
        # Format citations
        citations = self.format_citations(docs)
        
        return {
            "answer": response,
            "citations": citations,
            "confidence": self.calculate_confidence(docs, query),
            "source_breakdown": self.analyze_source_breakdown(docs)
        }
    
    def format_citations(self, docs: List[Document]) -> List[Dict[str, Any]]:
        """Format source citations"""
        citations = []
        
        for i, doc in enumerate(docs, 1):
            metadata = doc.metadata
            
            citation = {
                "id": f"ref_{i}",
                "title": metadata.get("title", "Unbekanntes Dokument"),
                "source_type": metadata.get("source_type", "unknown"),
                "source_path": metadata.get("source_path", ""),
                "excerpt": doc.page_content[:200] + "..." if len(doc.page_content) > 200 else doc.page_content,
                "relevance_score": metadata.get("relevance_score", 0),
                "page_number": metadata.get("page_number"),
                "section": metadata.get("section"),
                "last_updated": metadata.get("last_updated")
            }
            
            # Source-specific formatting
            if metadata.get("source_type") == "streamworks_help":
                citation["display_name"] = f"StreamWorks Help - {citation['title']}"
                citation["icon"] = "📚"
            elif metadata.get("source_type") == "jira":
                citation["display_name"] = f"JIRA - {citation['title']}"
                citation["icon"] = "🎫"
            elif metadata.get("source_type") == "ddds":
                citation["display_name"] = f"DDDS - {citation['title']}"
                citation["icon"] = "📊"
            else:
                citation["display_name"] = citation["title"]
                citation["icon"] = "📄"
            
            citations.append(citation)
        
        return citations
    
    def analyze_source_breakdown(self, docs: List[Document]) -> Dict[str, Any]:
        """Analyze which sources contributed to the answer"""
        source_counts = {}
        
        for doc in docs:
            source_type = doc.metadata.get("source_type", "unknown")
            source_counts[source_type] = source_counts.get(source_type, 0) + 1
        
        return {
            "total_sources": len(docs),
            "by_type": source_counts,
            "primary_source": max(source_counts.items(), key=lambda x: x[1])[0] if source_counts else None
        }
```

class ResponseFormatter:

        """Format response with inline citations"""

        formatted_answer = answer

                    formatted_answer += f"  📅 Aktualisiert: {citation['last_updated']}\n"
                formatted_answer += f"  📝 {citation['excerpt']}\n\n"
        
        return formatted_answer
    
    @staticmethod
    def format_for_user_role(answer: str, 
                           citations: List[Dict[str, Any]], 
                           user_role: UserRole) -> str:
        """Format response based on user role"""
        
        if user_role == UserRole.DEVELOPER:
            return ResponseFormatter.format_technical_response(answer, citations)
        elif user_role == UserRole.BUSINESS_USER:
            return ResponseFormatter.format_business_response(answer, citations)
        elif user_role == UserRole.SUPPORT:
            return ResponseFormatter.format_support_response(answer, citations)
        else:
            return ResponseFormatter.format_response_with_citations(answer, citations)
    
    @staticmethod
    def format_technical_response(answer: str, citations: List[Dict[str, Any]]) -> str:
        """Format for developers - include file paths and technical details"""
        formatted = answer + "\n\n**Technische Referenzen:**\n"
        
        for citation in citations:
            formatted += f"- **{citation['title']}**\n"
            formatted += f"  📂 Dateipfad: `{citation['source_path']}`\n"
            if citation.get('extra_metadata'):
                extra = citation['extra_metadata']
                if extra.get('module'):
                    formatted += f"  🔧 Modul: {extra['module']}\n"
                if extra.get('version'):
                    formatted += f"  📦 Version: {extra['version']}\n"
            formatted += "\n"
        
        return formatted
